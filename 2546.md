class Solution {
public:
    bool makeStringsEqual(string s, string target) {
        int n=s.size();
        if(s==target) return 1;
        
        int count1=0,count2=0;
        for(int i=0;i<n;i++){
            if(s[i]=='1') count1++;
            if(target[i]=='1') count2++;
        }
       if(!count1 or !count2) return 0;
          
        
        return 1;
    }
};
